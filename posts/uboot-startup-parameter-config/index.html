uboot启动参数配置

2017-06-16 One-minute read
uboot
uboot • param

uboot启动参数 Link to heading

1setenv boot_mtd 'bootz'
2setenv ipaddr   '192.168.20.126'
3setenv serverip '192.168.20.9'
4setenv netmask  '255.255.255.0'
5setenv mynfsroot '/home/yocto/hedong/nfs/rootfs_hd'
6setenv loadtftpfdt 'tftp ${fdt_addr} ${fdt_file}'
7setenv loadtftpimage 'tftp ${loadaddr} ${image}'
8setenv nfsboot 'echo Booting from net using tftp and nfs...; run nfsbootargs;run loadtftpimage; run loadtftpfdt;${boot_mtd} ${loadaddr} - ${fdt_addr}'
9setenv nfsbootargs 'setenv bootargs root=/dev/nfs rw ip=${ipaddr}:${serverip}::${netmask}::eth0:off nfsroot=${serverip}:${nfsroot},nolock'
